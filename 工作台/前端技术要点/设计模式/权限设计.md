## 核心代码

支持计算权限的运算：支持括号优先级 如： A&(B|C)

```jsx
function infixToPostFix(infix: string): string[] {

const postfix: string[] = []

const postfixHelper = new Stack<string>()

  

for (let i = 0; i < infix.length; i++) {

const c = infix[i]

if (!isOpera(c)) {

// 处理操作数

let operands = ''

while (i < infix.length && !isOpera(infix[i])) {

operands += infix[i++]

}

postfix.push(operands)

i--

} else {

// 处理操作符做出处理

handlerSymbol(c, postfixHelper, postfix)

}

}
```
